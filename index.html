function positiveNumbers(a, b, c)
{
let result = 0;
if (a > 0)
{
result += a;
}
if (b > 0)
{
result += b;
}
if (c > 0)
{
result += c;
}
return result;
}

функция (перевірена робоча, значення потрібно міняти вручну):

function getRandomInt(a,b,c)
{
let result;
let x = a * b * c;
let y = a + b + c;
if (x > y)
{
result = x;
}
else result = y;
return result + 3;
}
getRandomInt(10, 20,30);
